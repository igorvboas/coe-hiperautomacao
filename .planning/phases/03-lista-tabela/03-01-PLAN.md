```ts
import 'server-only';

import { createClient } from '@/lib/supabase/server';
import type { Opportunity, OpportunityKpis, OpportunityStatus } from './types';

/**
 * Busca todas as oportunidades visíveis pro tenant do usuário logado.
 * RLS filtra automaticamente — o backend não precisa passar tenant_id.
 *
 * Ordenação default: score desc, depois seq_id asc (paridade com sort default do mockup).
 */
export async function fetchOpportunities(): Promise<Opportunity[]> {
  const supabase = await createClient();

  const { data, error } = await supabase
    .from('opportunities_with_score')
    .select('*')
    .order('score', { ascending: false })
    .order('seq_id', { ascending: true });

  if (error) {
    // Server Component error → propaga pra error.tsx do Next
    throw new Error(`Erro ao buscar oportunidades: ${error.message}`);
  }

  return data ?? [];
}

/**
 * Calcula os KPIs a partir do array. Usado pela KPI bar.
 * Não vai ao banco — opera em memória sobre o array já carregado.
 */
export function computeKpis(opps: Opportunity[]): OpportunityKpis {
  const byStatus: Record<OpportunityStatus, number> = {
    novo: 0, em_analise: 0, planejamento: 0, backlog: 0,
    desenvolvimento: 0, homologacao: 0, producao: 0, concluido: 0,
  };
  const byTool = { rpa: 0, n8n: 0, ambos: 0 };
  const byPriority = { alta: 0, media: 0, baixa: 0 };

  let totalScore = 0;
  let scoreCount = 0;
  let personas = 0;
  let formularios = 0;

  for (const o of opps) {
    if (o.source === 'persona') personas++;
    else if (o.source === 'formulario') formularios++;

    byStatus[o.status] = (byStatus[o.status] ?? 0) + 1;

    if (o.ferramenta === 'rpa') byTool.rpa++;
    else if (o.ferramenta === 'n8n') byTool.n8n++;
    else if (o.ferramenta === 'ambos') byTool.ambos++;

    if (o.priority_level === 'alta') byPriority.alta++;
    else if (o.priority_level === 'media') byPriority.media++;
    else if (o.priority_level === 'baixa') byPriority.baixa++;

    if (typeof o.score === 'number') {
      totalScore += o.score;
      scoreCount++;
    }
  }

  return {
    total: opps.length,
    personas,
    formularios,
    scoreMedio: scoreCount > 0 ? Math.round(totalScore / scoreCount) : 0,
    byStatus,
    byTool,
    byPriority,
  };
}
```

**`import 'server-only'`** é o que impede esse arquivo de ser bundlado pro client por engano. Se algum dia um Client Component tentar importar, build quebra.
