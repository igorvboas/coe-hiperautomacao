```ts
import { z } from 'zod';

export const riskTypeEnum = z.enum(['impedimento', 'risco', 'oportunidade']);
export const riskImpactEnum = z.enum(['alto', 'significativo', 'moderado', 'baixo']);
export const riskProbabilityEnum = z.enum(['provavel', 'possivel', 'improvavel', 'remota']);
export const riskStatusEnum = z.enum(['novo', 'gerenciado', 'mitigado', 'ocorrido']);

// priority é GENERATED (trigger, matriz impacto×probabilidade) → NUNCA entra no input.
// id/tenant_id/opportunity_id são server-derived → rejeitados pelo .strict().
export const riskInputSchema = z
  .object({
    descricao: z.string().min(1, 'Descrição obrigatória').max(2000, 'Máximo 2000 caracteres'),
    tipo: riskTypeEnum,
    responsavel: z.string().max(200, 'Máximo 200 caracteres').optional().or(z.literal('')),
    impacto: riskImpactEnum,
    probabilidade: riskProbabilityEnum,
    status: riskStatusEnum.default('novo'),
    resposta: z.string().max(2000, 'Máximo 2000 caracteres').optional().or(z.literal('')),
    descricao_impacto: z.string().max(2000, 'Máximo 2000 caracteres').optional().or(z.literal('')),
  })
  .strict();

export type RiskInput = z.infer<typeof riskInputSchema>;
```

NÃO criar queries nem actions de risco (Phase 12). Este arquivo é puramente validação/tipos — satisfaz a parte de "validação" da dependência declarada da P12.
