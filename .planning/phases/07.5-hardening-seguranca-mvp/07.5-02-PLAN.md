    ```sql
    -- =============================================================================
    -- 0006_seq_id_atomic.sql — Atomicidade no seq_id por tenant
    -- =============================================================================
    -- Substitui o trigger `set_opportunity_seq_id()` original de 0001_init.sql,
    -- que usa `select max(seq_id) + 1` (race-prone), por uma versão atômica baseada
    -- em tabela auxiliar `tenant_sequences` + `UPDATE ... RETURNING` (row-lock
    -- transacional).
    --
    -- Defesa em profundidade:
    --   - Trigger SEMPRE sobrescreve `new.seq_id` — payload do cliente é ignorado.
    --   - tenant_sequences tem RLS habilitado e ZERO policies (só SECURITY DEFINER acessa).
    --
    -- Pré-requisitos: 0001_init.sql aplicado, opportunities populated ou não.
    -- =============================================================================

    set session characteristics as transaction read write;
    set default_transaction_read_only = off;
    set check_function_bodies = off;

    -- ---------------------------------------------------------------------------
    -- 1. Tabela auxiliar — counter por tenant
    -- ---------------------------------------------------------------------------

    create table if not exists tenant_sequences (
      tenant_id  uuid primary key references tenants(id) on delete cascade,
      last_seq   int  not null default 0
    );

    alter table tenant_sequences enable row level security;
    -- ZERO policies → role authenticated/anon não lê.
    -- Service role bypassa RLS por default.

    -- ---------------------------------------------------------------------------
    -- 2. Backfill — alinhar com o estado atual de opportunities
    -- ---------------------------------------------------------------------------

    -- 2a. Tenants COM oportunidades: last_seq = max(seq_id)
    insert into tenant_sequences (tenant_id, last_seq)
    select tenant_id, coalesce(max(seq_id), 0)
    from opportunities
    group by tenant_id
    on conflict (tenant_id) do update set last_seq = excluded.last_seq;

    -- 2b. Tenants SEM oportunidades: last_seq = 0
    insert into tenant_sequences (tenant_id, last_seq)
    select id, 0 from tenants
    on conflict (tenant_id) do nothing;

    -- ---------------------------------------------------------------------------
    -- 3. Função atômica — next_seq_id(p_tenant_id)
    -- ---------------------------------------------------------------------------

    create or replace function next_seq_id(p_tenant_id uuid)
    returns int
    language plpgsql
    security definer
    set search_path = public
    as $$
    declare
      v_next int;
    begin
      -- Upsert atômico — adquire row-lock no INSERT ON CONFLICT.
      -- Transações concorrentes para o mesmo tenant aguardam (pg_locks),
      -- não disputam o `max(seq_id)`.
      insert into tenant_sequences (tenant_id, last_seq)
        values (p_tenant_id, 1)
      on conflict (tenant_id) do update
        set last_seq = tenant_sequences.last_seq + 1
      returning last_seq into v_next;
      return v_next;
    end;
    $$;

    -- ---------------------------------------------------------------------------
    -- 4. Substitui o trigger antigo
    -- ---------------------------------------------------------------------------
    -- Pitfall 5 (RESEARCH.md): drop ANTES de recriar — `create or replace function`
    -- substitui função; `create trigger` SEM drop cria duplicado → comportamento
    -- indeterminado.

    drop trigger if exists trg_opportunities_seq_id on opportunities;
    drop function if exists set_opportunity_seq_id();

    create or replace function set_opportunity_seq_id()
    returns trigger
    language plpgsql
    as $$
    begin
      -- SEMPRE sobrescreve — bloqueia cliente que tente forjar seq_id.
      new.seq_id := next_seq_id(new.tenant_id);
      return new;
    end;
    $$;

    create trigger trg_opportunities_seq_id
      before insert on opportunities
      for each row execute function set_opportunity_seq_id();

    -- =============================================================================
    -- FIM 0006 — seq_id agora é atômico e cliente não pode forjar
    -- =============================================================================
    ```

    - `drop trigger if exists` e `drop function if exists` ANTES dos `create`.

    - RLS habilitado em `tenant_sequences` sem nenhuma policy criada.
