function onSave() {
  // Pitfall 2: gate the all-or-null criterios before submit
  const cv = validateStep('criterios', form);
  if (!cv.ok) { setErrors(cv.errors); return; }
  setSubmitError(null);
  startTransition(async () => {
    const payload = {
      ...form,
      prioridade_fte:
        form.fte_horas != null ? deriveFteBucket(Number(form.fte_horas)) : undefined,
    };
    const result = await updateOpportunity(opportunity.id, payload as OpportunityInput);
    if (!result.ok) {
      setSubmitError(result.error);
      if (result.fieldErrors) setErrors(result.fieldErrors);
      return;
    }
    setEditMode(false);
    router.refresh();   // Pitfall 4: modal stays open, repaints DB-authoritative values
  });
}
function onCancel() { setForm(opportunityToFormData(opportunity)); setErrors({}); setSubmitError(null); setEditMode(false); }

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-13-05-01 | Elevation/Tampering (mass assignment) | client edit payload → `updateOpportunity` | mitigate | Submit ONLY the `WizardFormData` shape (via `opportunityToFormData`) + the derived `prioridade_fte`. `opportunityInputSchema.strict()` rejects `unrecognized_keys` (tenant_id/id/seq_id/score/rpa_score/priority_level); the action enumerates `.update({...})` explicitly. | T-13-05-04 | Tampering (DB CHECK bypass) | partial criterios save | mitigate | `validateStep('criterios', form)` gates the all-or-null criterios before submit; Zod `.refine` + DB CHECK `opportunities_criterios_chk` are the server-side backstops. |
