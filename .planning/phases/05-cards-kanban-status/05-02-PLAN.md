  <action>
```ts
'use server';

import { revalidatePath } from 'next/cache';
import { createClient } from '@/lib/supabase/server';
import type { OpportunityStatus } from './types';

export type UpdateStatusResult = { ok: true } | { ok: false; error: string };

export async function updateOpportunityStatus(
  id: string,
  newStatus: OpportunityStatus
): Promise<UpdateStatusResult> {
  const supabase = await createClient();
  const { error } = await supabase
    .from('opportunities')
    .update({ status: newStatus })
    .eq('id', id);

  if (error) {
    return { ok: false, error: `Falha ao atualizar status: ${error.message}` };
  }

  revalidatePath('/opportunities');
  revalidatePath(`/opportunities/${id}`);
  return { ok: true };
}
```

RLS bloqueia mudança de oportunidade de outro tenant — não precisa filtrar manualmente.
