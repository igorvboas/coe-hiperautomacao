```tsx
export default async function OpportunitiesPage({
  searchParams,
}: {
  searchParams: Promise<{ view?: string }>;
}) {
  const { view } = await searchParams;
  const opportunities = await fetchOpportunities();
  const kpis = computeKpis(opportunities);

  return (
    <div className="flex flex-col min-h-full">
      <KpiBar kpis={kpis} />
      <Toolbar counts={{ visible: opportunities.length, total: opportunities.length }} />
      <div className="flex-1 px-6 py-4">
        {view === 'cards' ? (
          <OpportunityCards opportunities={opportunities} />
        ) : view === 'kanban' ? (
          <Placeholder text="Kanban virá no plan 05-03" />
        ) : (
          <OpportunityTable opportunities={opportunities} />
        )}
      </div>
    </div>
  );
}
```

`<Placeholder>` é só um div inline; vai ser substituído no plan 05-03.
